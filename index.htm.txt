    <script>
        let myChart = null;

        function calculate() {
            // 讀取輸入值
            const homePrice = parseFloat(document.getElementById('homePrice').value) * 10000;
            const downPayment = parseFloat(document.getElementById('downPayment').value) * 10000;
            const loanRate = parseFloat(document.getElementById('loanInterest').value) / 100;
            const loanYears = parseInt(document.getElementById('loanYears').value);
            const homeAppreciation = parseFloat(document.getElementById('homeAppreciation').value) / 100;
            
            const startRent = parseFloat(document.getElementById('monthlyRent').value);
            const rentIncrease = parseFloat(document.getElementById('rentIncrease').value) / 100;
            const investReturn = parseFloat(document.getElementById('investReturn').value) / 100;

            const loanAmount = homePrice - downPayment;
            const monthlyRate = loanRate / 12;
            const totalMonths = loanYears * 12;

            // 計算房貸 (本息平均攤還)
            let monthlyMortgage = 0;
            if (loanRate > 0) {
                monthlyMortgage = (loanAmount * monthlyRate * Math.pow(1 + monthlyRate, totalMonths)) / (Math.pow(1 + monthlyRate, totalMonths) - 1);
            } else {
                monthlyMortgage = loanAmount / totalMonths;
            }

            document.getElementById('monthlyMortgageDisplay').innerText = 'NT$ ' + Math.round(monthlyMortgage).toLocaleString();

            // 數據計算容器
            let years = [];
            let buyerNetWorth = [];
            let renterNetWorth = [];
            let tableRows = '';

            let currentHomeValue = homePrice;
            let currentLoanBalance = loanAmount;
            let currentRenterInvest = downPayment; // 租房者將頭期款拿去投資
            let currentMonthlyRent = startRent;

            for (let year = 0; year <= 40; year++) {
                if (year > 0) {
                    // 1. 房屋增值與房貸還款
                    currentHomeValue *= (1 + homeAppreciation);
                    
                    for(let m = 0; m < 12; m++) {
                        let interest = currentLoanBalance * monthlyRate;
                        let principal = monthlyMortgage - interest;
                        if (currentLoanBalance > 0) {
                            currentLoanBalance -= principal;
                        }
                    }
                    if (currentLoanBalance < 0) currentLoanBalance = 0;

                    // 2. 租房端投資成長
                    // 投資複利
                    currentRenterInvest *= (1 + investReturn);
                    
                    // 差額投資：如果 (房貸 > 租金)，把省下的錢投資；反之，若 (租金 > 房貸)，需從投資中扣除差額
                    let annualDiff = (monthlyMortgage - currentMonthlyRent) * 12;
                    currentRenterInvest += annualDiff;

                    // 租金漲幅
                    currentMonthlyRent *= (1 + rentIncrease);
                }

                const buyerNet = (currentHomeValue - currentLoanBalance) / 10000;
                const renterNet = currentRenterInvest / 10000;

                years.push(year);
                buyerNetWorth.push(Math.round(buyerNet));
                renterNetWorth.push(Math.round(renterNet));

                // 更新摘要 (以30年為例)
                if (year === 30) {
                    document.getElementById('buyerWealthDisplay').innerText = Math.round(buyerNet).toLocaleString() + ' 萬';
                    document.getElementById('renterWealthDisplay').innerText = Math.round(renterNet).toLocaleString() + ' 萬';
                }

                // 生成表格列 (每5年顯示一次或重要節點)
                if (year % 5 === 0 || year === 40) {
                    tableRows += `
                        <tr class="border-b hover:bg-gray-50">
                            <td class="px-4 py-2 font-medium">第 ${year} 年</td>
                            <td class="px-4 py-2">${Math.round(currentHomeValue/10000).toLocaleString()} 萬</td>
                            <td class="px-4 py-2">${Math.round(currentLoanBalance/10000).toLocaleString()} 萬</td>
                            <td class="px-4 py-2 font-semibold text-blue-600">${Math.round(buyerNet).toLocaleString()} 萬</td>
                            <td class="px-4 py-2 font-semibold text-green-600">${Math.round(renterNet).toLocaleString()} 萬</td>
                            <td class="px-4 py-2">${Math.round(currentMonthlyRent).toLocaleString()} 元</td>
                        </tr>
                    `;
                }
            }

            document.getElementById('dataTable').innerHTML = tableRows;
            renderChart(years, buyerNetWorth, renterNetWorth);
        }

        function renderChart(labels, buyerData, renterData) {
            const ctx = document.getElementById('comparisonChart').getContext('2d');
            
            if (myChart) {
                myChart.destroy();
            }

            myChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels.map(y => '第' + y + '年'),
                    datasets: [
                        {
                            label: '買房資產淨值',
                            data: buyerData,
                            borderColor: '#3b82f6',
                            backgroundColor: 'rgba(59, 130, 246, 0.1)',
                            fill: true,
                            tension: 0.3
                        },
                        {
                            label: '租房投資資產',
                            data: renterData,
                            borderColor: '#10b981',
                            backgroundColor: 'rgba(16, 185, 129, 0.1)',
                            fill: true,
                            tension: 0.3
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        tooltip: {
                            mode: 'index',
                            intersect: false
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            title: {
                                display: true,
                                text: '淨值 (萬元)'
                            }
                        }
                    }
                }
            });
        }

        // 初始載入
        window.onload = calculate;
    </script>
